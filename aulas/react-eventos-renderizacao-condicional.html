React – Eventos, Formulários e Navegação

Nesta aula vamos trabalhar com eventos (onClick, onChange, etc.), formulários controlados, renderização condicional, validações e navegação entre páginas com React Router. Vamos construir uma aplicação completa de cadastro de contatos usando Next.js e Tailwind.

  • ✅ Projeto prático completo

1. Conceitos Fundamentais

1.1 Eventos no React

Eventos em React são passados como props para elementos. Use funções para tratar ações do usuário. 

export default function ExemploEventos() {
  function handleClick() {
    alert('Botão clicado!');
  }

  function handleChange(e) {
    console.log('Valor:', e.target.value);
  }

  return (
    <div>
      <button onClick={handleClick}>Clique aqui</button>
      <input onChange={handleChange} placeholder="Digite algo" />
    </div>
  );
}

1.2 Formulários Controlados

Em formulários controlados, o valor dos inputs é controlado pelo estado do componente via useState. 

import { useState } from 'react';

export default function FormControlado() {
  const [nome, setNome] = useState('');

  function handleSubmit(e) {
    e.preventDefault();
    alert(`Olá, ${nome}!`);
  }

  return (
    <form onSubmit={handleSubmit}>
      <input value={nome} onChange={(e) => setNome(e.target.value)} />
      <button type="submit">Enviar</button>
    </form>
  );
}

2. Projeto Prático: Cadastro de Contatos

2.1 Estrutura da Aplicação

Vamos criar uma aplicação completa de cadastro de contatos com navegação entre páginas. A estrutura será:

  • Página Principal: Lista de contatos com formulário de cadastro
  • Página de Detalhes: Exibe informações completas de um contato
  • Componentes: Formulário, lista, filtro e item de contato
Estrutura de Pastas Completa
app/
├── page.js                    # Página inicial (/)
├── layout.js                  # Layout compartilhado
├── globals.css                # Estilos globais com Tailwind
├── contact/
│   └── [id]/
│       └── page.js            # Página de detalhes (/contact/[id])
└── components/
    ├── FilterInput.jsx        # Componente de filtro
    ├── ContactForm.jsx        # Formulário de contato
    ├── ContactList.jsx        # Lista de contatos
    └── ContactItem.jsx        # Item individual de contato
Nota: A pasta components é opcional, mas ajuda a organizar melhor o código.
Layout Compartilhado (app/layout.js)

O layout compartilhado é aplicado a todas as páginas da aplicação e deve importar o arquivo globals.css para que o Tailwind CSS seja aplicado: 

// app/layout.js
import './globals.css'

export default function RootLayout({ children }) {
  return (
    <html lang="pt-BR">
      <body>
        <main>{children}</main>
      </body>
    </html>
  );
}
Arquivo de Estilos Globais (app/globals.css)

O arquivo globals.css deve conter as diretivas do Tailwind CSS:

/* app/globals.css */
@import "tailwindcss";
Importante: Sem a importação do globals.css no layout e as diretivas do Tailwind no globals.css, as classes do Tailwind não funcionarão na aplicação.
Configuração do Tailwind CSS

Para que o Tailwind CSS funcione corretamente, você também precisa ter o arquivo de configuração:

2.2 Página Principal (app/page.js)

Esta é a página inicial da aplicação. Ela gerencia o estado global dos contatos e integra todos os componentes. Quando um contato é clicado na lista, o usuário é direcionado para a página de detalhes. 

"use client";
import { useState, useEffect } from "react";
import { FilterInput } from "./components/FilterInput";
import { ContactForm } from "./components/ContactForm";
import { ContactList } from "./components/ContactList";

export default function HomePage() {
  const [contacts, setContacts] = useState([]);
  const [filter, setFilter] = useState("");

  // Carrega contatos do localStorage ao iniciar
  useEffect(() => {
    const savedContacts = JSON.parse(localStorage.getItem('contacts') || '[]');
    setContacts(savedContacts);
  }, []);

  // Salva contatos no localStorage sempre que mudar
  useEffect(() => {
    localStorage.setItem('contacts', JSON.stringify(contacts));
  }, [contacts]);

  function handleSubmit(newContact) {
    setContacts(prev => [...prev, newContact]);
  }

  function handleRemove(id) {
    setContacts(prev => prev.filter(c => c.id !== id));
  }

  const filtered = contacts.filter(c =>
    c.nome.toLowerCase().includes(filter.toLowerCase()) ||
    c.email.toLowerCase().includes(filter.toLowerCase())
  );

  return (
    <div className="min-h-screen bg-gray-50 p-6">
      <div className="max-w-3xl mx-auto space-y-6">
        <header className="flex items-center justify-between">
          <h1 className="text-2xl font-bold text-gray-900">Cadastro de Contatos</h1>
          <FilterInput value={filter} onChange={setFilter} />
        </header>

        <ContactForm onAdd={handleSubmit} />

        <ContactList items={filtered} onRemove={handleRemove} />
      </div>
    </div>
  );
}
Integração: Esta página salva os contatos no localStorage, que serão acessados pela página de detalhes quando um contato for clicado.

ContactForm Component
import { useState } from "react";

export function ContactForm({ onAdd }) {
  const [form, setForm] = useState({ nome: "", email: "", telefone: "" });
  const [errors, setErrors] = useState({});

  function validate(values) {
    const newErrors = {};
    if (!values.nome.trim()) newErrors.nome = "Informe o nome";
    if (!/^[^@\s]+@[^@\s]+\.[^@\s]+$/.test(values.email)) newErrors.email = "Email inválido";
    if (!/^\d{10,11}$/.test(values.telefone)) newErrors.telefone = "Telefone deve ter 10-11 dígitos";
    return newErrors;
  }

  function handleChange(e) {
    const { name, value } = e.target;
    setForm((prev) => ({ ...prev, [name]: value }));
  }

  function handleSubmit(e) {
    e.preventDefault();
    const v = validate(form);
    setErrors(v);
    if (Object.keys(v).length) return;
    onAdd({ ...form, id: Date.now() });
    setForm({ nome: "", email: "", telefone: "" });
  }

  return (
    <form onSubmit={handleSubmit} className="bg-white shadow rounded p-4 space-y-4">
      <div>
        <label className="block text-sm font-medium mb-1 text-gray-700">Nome</label>
        <input
          name="nome"
          className="w-full border rounded px-3 py-2 text-gray-900"
          value={form.nome}
          onChange={handleChange}
        />
        {errors.nome && <p className="text-sm text-red-600 mt-1">{errors.nome}</p>}
      </div>

      <div>
        <label className="block text-sm font-medium mb-1 text-gray-700">Email</label>
        <input
          name="email"
          className="w-full border rounded px-3 py-2 text-gray-900"
          value={form.email}
          onChange={handleChange}
        />
        {errors.email && <p className="text-sm text-red-600 mt-1">{errors.email}</p>}
      </div>

      <div>
        <label className="block text-sm font-medium mb-1 text-gray-700">Telefone</label>
        <input
          name="telefone"
          className="w-full border rounded px-3 py-2 text-gray-900"
          value={form.telefone}
          onChange={handleChange}
        />
        {errors.telefone && <p className="text-sm text-red-600 mt-1">{errors.telefone}</p>}
      </div>

      <div className="flex gap-2">
        <button type="submit" className="bg-blue-600 hover:bg-blue-700 text-white px-4 py-2 rounded">
          Adicionar
        </button>
        <button type="button" className="bg-gray-100 hover:bg-gray-200 text-gray-900 px-4 py-2 rounded" onClick={() => setForm({ nome: '', email: '', telefone: '' })}>
          Limpar
        </button>
      </div>
    </form>
  );
}

3. Next.js App Router - Navegação entre Páginas

3.1 Estrutura de Pastas do App Router

O Next.js App Router usa uma estrutura baseada em pastas. Cada pasta representa uma rota: 

app/
├── page.js              # Página inicial (/)
├── contact/
│   └── [id]/
│       └── page.js      # Página de detalhes (/contact/[id])
└── layout.js            # Layout compartilhado
Nota: O [id] na pasta /contact/[id] é um parâmetro dinâmico que será capturado pelo componente da página.

3.2 Componente Link do Next.js

O componente Link do Next.js permite navegação sem recarregar a página, mantendo a experiência SPA (Single Page Application). 

import Link from 'next/link';

// Exemplo de uso no ContactItem
<Link 
  href={`/contact/${contact.id}`} 
  className="font-medium hover:text-blue-600 transition-colors"
>
  {contact.nome}
</Link>
Importante: Sempre use Link do Next.js em vez de <a> para navegação interna. O <a> recarrega a página, perdendo o estado da aplicação.

3.3 Página de Detalhes do Contato (app/contact/[id]/page.js)

Esta página é acessada quando um contato é clicado na lista principal. Ela busca o contato específico no localStorage usando o ID da URL e exibe seus detalhes. 

// app/contact/[id]/page.js
"use client";
import { useParams, useRouter } from 'next/navigation';
import { useState, useEffect } from 'react';

export default function ContactDetailPage() {
  const params = useParams(); // Captura os parâmetros da URL
  const router = useRouter(); // Hook para navegação programática
  const [contact, setContact] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // Busca o contato do localStorage (mesmo local onde a página principal salva)
    const contacts = JSON.parse(localStorage.getItem('contacts') || '[]');
    const foundContact = contacts.find(c => c.id === parseInt(params.id));
    
    if (foundContact) {
      setContact(foundContact);
    }
    setLoading(false);
  }, [params.id]);

  function handleBack() {
    router.push('/'); // Volta para a página principal
  }

  if (loading) {
    return (
      <div className="min-h-screen bg-gray-50 flex items-center justify-center">
        <div className="text-lg text-gray-900">Carregando...</div>
      </div>
    );
  }

  if (!contact) {
    return (
      <div className="min-h-screen bg-gray-50 flex items-center justify-center">
        <div className="text-center">
          <h2 className="text-xl font-bold mb-4 text-gray-900">Contato não encontrado</h2>
          <button 
            onClick={handleBack}
            className="bg-blue-600 hover:bg-blue-700 text-white px-4 py-2 rounded"
          >
            Voltar
          </button>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-50 p-6">
      <div className="max-w-2xl mx-auto">
        <header className="mb-6">
          <button 
            onClick={handleBack}
            className="text-blue-600 hover:text-blue-700 mb-4 flex items-center"
          >
            <span className="mr-2">←</span> Voltar
          </button>
          <h1 className="text-3xl font-bold text-gray-900">Detalhes do Contato</h1>
        </header>

        <div className="bg-white shadow rounded-lg p-6">
          <div className="space-y-4">
            <div>
              <label className="block text-sm font-medium text-gray-700">Nome</label>
              <p className="text-lg text-gray-900">{contact.nome}</p>
            </div>
            
            <div>
              <label className="block text-sm font-medium text-gray-700">Email</label>
              <p className="text-lg text-gray-900">{contact.email}</p>
            </div>
            
            <div>
              <label className="block text-sm font-medium text-gray-700">Telefone</label>
              <p className="text-lg text-gray-900">{contact.telefone}</p>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}
Fluxo de Integração:
  1. Usuário clica em um contato na página principal
  2. Next.js navega para /contact/[id]
  3. Página de detalhes busca o contato no localStorage
  4. Exibe os detalhes completos do contato
  5. Botão "Voltar" retorna à página principal

3.4 Passagem de Parâmetros via URL

Existem várias formas de passar dados entre páginas via URL:

Parâmetros de Rota ([id])
// Pasta: /contact/[id]
// URL: /contact/123
// Captura: const params = useParams(); // params.id = "123"

<Link href="/contact/123">Ver Contato</Link>
Query Parameters (?search=termo)
// URL: /contacts?search=joao&filter=active
import { useSearchParams } from 'next/navigation';

function ContactList() {
  const searchParams = useSearchParams();
  const search = searchParams.get('search'); // "joao"
  const filter = searchParams.get('filter'); // "active"

  function updateSearch(newSearch) {
    const params = new URLSearchParams(searchParams);
    params.set('search', newSearch);
    router.push(`/contacts?${params.toString()}`);
  }
}
Navegação Programática
import { useRouter } from 'next/navigation';

function LoginForm() {
  const router = useRouter();

  function handleLogin() {
    // Lógica de login...
    if (loginSuccessful) {
      router.push('/dashboard'); // Navega para dashboard

}

Exercício 1: Formulário de Login com Navegação

Objetivo: Criar um formulário de login que navega para uma página de dashboard após sucesso.

  1. Crie campos de email e senha controlados
  2. Valide email (regex) e senha (mín. 6 chars)
  3. Mostre mensagens de erro condicionalmente
  4. Após login bem-sucedido, navegue para /dashboard
Mostrar solução 
"use client";
import { useState } from "react";
import { useRouter } from 'next/navigation';

export default function LoginForm() {
  const [form, setForm] = useState({ email: "", senha: "" });
  const [errors, setErrors] = useState({});
  const router = useRouter();

  function validate(v) {
    const e = {};
    if (!/^[^@\s]+@[^@\s]+\.[^@\s]+$/.test(v.email)) e.email = "Email inválido";
    if ((v.senha || '').length < 6) e.senha = "Mínimo 6 caracteres";
    return e;
  }

  function handleSubmit(ev) {
    ev.preventDefault();
    const e = validate(form);
    setErrors(e);
    if (Object.keys(e).length) return;
    
    // Simula login bem-sucedido
    router.replace('/dashboard');
  }

  return (
    <form onSubmit={handleSubmit} className="space-y-3">
      <div>
        <label className="block text-sm">Email</label>
        <input
          className="border rounded px-3 py-2 w-full"
          value={form.email}
          onChange={(e) => setForm((p) => ({ ...p, email: e.target.value }))}
        />
        {errors.email && <p className="text-sm text-red-600">{errors.email}</p>}
      </div>
      <div>
        <label className="block text-sm">Senha</label>
        <input
          type="password"
          className="border rounded px-3 py-2 w-full"
          value={form.senha}
          onChange={(e) => setForm((p) => ({ ...p, senha: e.target.value }))}
        />
        {errors.senha && <p className="text-sm text-red-600">{errors.senha}</p>}
      </div>
      <button className="bg-blue-600 hover:bg-blue-700 text-white px-4 py-2 rounded">
        Entrar
      </button>
    </form>
  );
}
